
import 'package:i_profile/core/utils/style.dart';
import 'package:i_profile/ui/pages/home_view.dart';
import 'package:rive/rive.dart';
import 'package:flutter/material.dart';
import 'package:flutter/services.dart';

class LoginPage extends StatefulWidget {
  const LoginPage({Key key}) : super(key: key);

  @override
  _LoginPageState createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
  bool get isPlaying => _controller?.isActive ?? false;

  Artboard _riveArtboard;
  RiveAnimationController _controller;
  @override
  void initState() {
    super.initState();
    riveLoading();
  }

  void riveLoading() {
    rootBundle.load('assets/rive/tealcolorleaf.riv').then(
      (data) async {
        final file = RiveFile();

        if (file.import(data)) {
          final artboard = file.mainArtboard;

          artboard.addController(_controller = SimpleAnimation('loading'));
          setState(() => _riveArtboard = artboard);
        }
      },
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Container(
        child: _controller?.isActive ==true ? _loadingSplash(): ,
      ),
      floatingActionButton: FloatingActionButton.extended(
        backgroundColor: primaryColor,
        foregroundColor: gray1,
        onPressed: () {
          
        },
        icon: Icon(Icons.keyboard_arrow_right),
        label: Text('LOGIN WITH FACEBOOK'),
      ),
    );
  }

  Widget _loadingSplash() {
    return _riveArtboard == null
        ? const SizedBox()
        : Rive(
            artboard: _riveArtboard,
            fit: BoxFit.fill,
          );
  }
}
